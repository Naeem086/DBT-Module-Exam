#1. Procedure Solution : 

CREATE TABLE TEMPP (
	OPERATION_NO INT auto_increment primary KEY,
	OPERATION VARCHAR(30)
);

DELIMITER //
CREATE PROCEDURE PROC1(IN STR1 VARCHAR(10), IN STR2 VARCHAR(10))
BEGIN 
	IF (LOCATE(STR1, STR2) > 0) THEN 
	INSERT INTO TEMPP (OPERATION) VALUES ('String Found');
	ELSE INSERT INTO TEMPP(OPERATION) VALUES ('No String Found');
	END IF;    

END; //
DELIMITER ;

#2. Function Solution :

DELIMITER //
CREATE FUNCTION FUNC1(X INT, Y INT, Z INT) RETURNS BOOLEAN DETERMINISTIC
BEGIN 
	IF NOT ((X+Y >= Z) AND (Y+Z >= X) AND (Z+X >= Y)) THEN RETURN FALSE;
	ELSE RETURN TRUE;
      END IF;
END; //
DELIMITER ;
